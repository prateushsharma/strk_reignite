import redis
import json

def candle_generator(redis_host='localhost', redis_port=6379, channel='binance_data'):
    """
    Generator that perfectly matches your publisher's data format:
    {
        "timestamp": str,
        "candlesticks": [
            {
                "timestamp": str,
                "open": float,
                "high": float,
                "low": float,
                "close": float,
                "volume": float, 
                "close_time": str,
                "quote_asset_volume": float,

            },
            ...
        ]
    }
    """
    r = redis.Redis(host=redis_host, port=redis_port)
    pubsub = r.pubsub()
    pubsub.subscribe(channel)
    
    # Wait for subscription confirmation
    while True:
        msg = pubsub.get_message()
        if msg and msg['type'] == 'subscribe':
            print(f"âœ… Successfully subscribed to {channel}")
            break
    
    try:
        while True:
            message = pubsub.get_message(timeout=0.5)
            
            if message and message['type'] == 'message':
                try:
                    data = json.loads(message['data'])
                    print(f"ğŸ“© Received batch at {data['timestamp']}")

                    yield data

                except Exception as e:
                    print(f"âš ï¸ Error processing message: {e}")

        pubsub.close()
        print("ğŸ”´ Redis connection closed")
